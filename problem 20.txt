#include <stdio.h>

int main(){
    char ch;
    scanf(" %c",&ch);
    int a ;
    scanf("%d",&a);
    int b;
    scanf("%d",&b);
    
    switch(ch){
    case '+':
        printf("%d",a+b);
        break;
    case '-':
       printf("%d",a-b);
        break;
    case '*':
        printf("%d",a*b);
        break;
    case '/':
        printf("%d",a/b);
        break;
    case '%':
        printf("%d",a%b);
        break;
    }
    return 0;
}
