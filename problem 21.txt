#include <stdio.h>

int main(){
    int ch;
    scanf(" %d",&ch);
    
    switch(ch){
    case 1:
        printf("monday");
        break;
    case 2:
       printf("Tuesday");
        break;
    case 3:
        printf("Wednesday");
        break;
    case 4:
        printf("Thursday");
        break;
    case 5:
        printf("friday");
        break;
    }
    return 0;
}
